func Tokenize(text []rune) ([]TokenSN, error) {
	gts := gner.Tokenize(text, NewTokenSN)
	res := make([]TokenSN, len(gts))
	for i := range gts {
		if t, ok := gts[i].(TokenSN); !ok {
			return nil, errors.New("Wrong token object")
		} else {
			res[i] = t
		}
	}
	return res, nil
}
